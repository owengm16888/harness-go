### 3. 流式输出

```go
// 流式输出面试过程
func streamInterview(ctx context.Context, instanceID string) {
    manager := routine.NewRoutineManager(config)
    
    // 启动面试
    manager.Start(ctx, instanceID)
    
    // 流式输出
    for {
        // 获取当前状态
        instance, _ := manager.GetInstance(ctx, instanceID)
        
        if instance.Status != routine.StatusRunning {
            break
        }
        
        // 获取问题
        question, _ := manager.GetQuestion(ctx, instanceID)
        if question != "" {
            fmt.Printf("\n[面试官] %s\n", question)
        }
        
        // 读取用户输入
        fmt.Print("\n[你的回答] ")
        var answer string
        reader := bufio.NewReader(os.Stdin)
        answer, _ = reader.ReadString('\n')
        answer = strings.TrimSpace(answer)
        
        // 提交回答
        manager.SubmitAnswer(ctx, instanceID, answer)
        
        // 显示评分
        instance, _ = manager.GetInstance(ctx, instanceID)
        if len(instance.Scores) > 0 {
            lastScore := instance.Scores[len(instance.Scores)-1]
            fmt.Printf("\n[评分] %.1f/100\n", lastScore.Score.Total)
        }
    }
    
    // 获取最终报告
    report, _ := manager.GetReport(ctx, instanceID)
    fmt.Println(report)
}
```

### 5. 面试数据分析

```go
// 分析面试数据
func analyzeInterviews(instances []*routine.RoutineInstance) {
    // 统计分数分布
    scoreDist := map[string]int{
        "excellent": 0, // 90-100
        "good":      0, // 70-89
        "average":   0, // 50-69
        "poor":      0, // <50
    }
    
    for _, instance := range instances {
        avg := instance.GetAverageScore()
        
        switch {
        case avg.Total >= 90:
            scoreDist["excellent"]++
        case avg.Total >= 70:
            scoreDist["good"]++
        case avg.Total >= 50:
            scoreDist["average"]++
        default:
            scoreDist["poor"]++
        }
    }
    
    fmt.Println("分数分布:")
    for level, count := range scoreDist {
        fmt.Printf("  %s: %d\n", level, count)
    }
    
    // 统计通过率
    passCount := 0
    for _, instance := range instances {
        if instance.FinalReport != nil && instance.FinalReport.Pass {
            passCount++
        }
    }
    
    passRate := float64(passCount) / float64(len(instances)) * 100
    fmt.Printf("\n通过率: %.1f%%\n", passRate)
    
    // 统计平均轮次
    totalRounds := 0
    for _, instance := range instances {
        totalRounds += instance.Round
    }
    avgRounds := float64(totalRounds) / float64(len(instances))
    fmt.Printf("平均轮次: %.1f\n", avgRounds)
}
```

```go
// SQLite 存储实现
type SQLiteRoutineStore struct {
    db *sql.DB
}

func (s *SQLiteRoutineStore) Save(ctx context.Context, instance *routine.RoutineInstance) error {
    data, _ := json.Marshal(instance)
    
    _, err := s.db.ExecContext(ctx,
        `INSERT OR REPLACE INTO routines (id, name, type, status, data, created_at) 
         VALUES (?, ?, ?, ?, ?, ?)`,
        instance.ID,
        instance.Config.Name,
        instance.Config.Type,
        instance.Status,
        string(data),
        instance.StartTime,
    )
    
    return err
}

func (s *SQLiteRoutineStore) Get(ctx context.Context, id string) (*routine.RoutineInstance, error) {
    var data string
    
    err := s.db.QueryRowContext(ctx,
        `SELECT data FROM routines WHERE id = ?`, id,
    ).Scan(&data)
    
    if err != nil {
        return nil, err
    }
    
    var instance routine.RoutineInstance
    json.Unmarshal([]byte(data), &instance)
    
    return &instance, nil
}
```
